import pandas as pd
import altair as alt
from palmerpenguins import load_penguins
from IPython.display import HTML

penguins = load_penguins()
def display_centered_chart(c):
    with alt.renderers.enable("default"):
        html = c._repr_mimebundle_()["text/html"]

    s = html.replace("display: flex;", "display: flex;\n    justify-content: center;")

    display(HTML(s))
c = alt.Chart(penguins).mark_point().encode(
    x=alt.X('bill_length_mm'),
    y=alt.Y('bill_depth_mm')
)

display_centered_chart(c)
